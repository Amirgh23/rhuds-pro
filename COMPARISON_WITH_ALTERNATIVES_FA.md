## 🔗 ترکیب استفاده

### RHUDS Pro + Chart.js

```tsx
// بهترین ترکیب برای پروژه‌های بزرگ
import { HudButton, CyberCard } from '@rhuds/components';
import { Chart } from 'chart.js';

export function Dashboard() {
  return (
    <CyberCard title="داشبورد">
      <HudButton>دکمه</HudButton>
      <Chart type="line" data={data} />
    </CyberCard>
  );
}
```

### RHUDS Pro + Arwes

```tsx
// ترکیب برای طراحی‌های خاص
import { HudButton } from '@rhuds/components';
import { Arwes } from 'arwes';

export function SpecialDesign() {
  return (
    <Arwes>
      <HudButton>دکمه</HudButton>
    </Arwes>
  );
}
```
